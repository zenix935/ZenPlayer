cmake_minimum_required(VERSION 3.16)
project(ZenPlayer VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Automated Qt pipeline handling
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

# Find your local Qt setup (adjust to Qt5 if necessary)
find_package(Qt6 REQUIRED COMPONENTS Widgets)

add_executable(ZenPlayer ${SOURCES})

# Define our executable and its sources
qt_add_executable(QtVsCodeApp 
    main.cpp
)

# Link against the Qt Widgets module
target_link_libraries(QtVsCodeApp PRIVATE Qt6::Widgets)
