cmake_minimum_required(VERSION 3.16)
project(ZenPlayer VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets Multimedia)

qt_add_executable(ZenPlayer
    main.cpp
    ZenPlayer.cpp
    ZenPlayer.h
    ZenPlayer.ui
    ZenPlayer.qrc
    removeDialog.h
    removeDialog.ui
    createPlaylistDialog.h
    createPlaylistDialog.ui
    addToPlaylistDialog.h
    addToPlaylistDialog.ui
)

target_link_libraries(ZenPlayer PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
    Qt6::Multimedia
)

target_include_directories(ZenPlayer PRIVATE
    "C:/MINE/codes/ZenPlayer/utils"
)
